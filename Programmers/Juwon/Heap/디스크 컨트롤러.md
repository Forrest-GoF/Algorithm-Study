```java
import java.io.*;
import java.util.*;

class Job implements Comparable<Job>{
    int requestTime;
    int requiredTime;

    Job(int requestTime, int requiredTime){
        this.requestTime = requestTime;
        this.requiredTime = requiredTime;
    }

    @Override
    public int compareTo(Job o) {
        return requiredTime - o.requiredTime;
    }
}

class Solution {
    public int solution(int[][] jobs) {
        int answer = 0;
        int currentTime = 0;
        Queue<Job> unAdmittedJobQueue = new LinkedList<>();
        PriorityQueue<Job> readyJobQueue = new PriorityQueue<>();

        Arrays.sort(jobs, (job1, job2) -> {
            if(job1[0] != job2[0]){
                return job1[0] - job2[0];
            }
            return job1[1] - job2[1];
        });

        for(int i = 0; i < jobs.length; i++){
            int requestTIme = jobs[i][0];
            int requiredTime = jobs[i][1];
            unAdmittedJobQueue.add(new Job(requestTIme, requiredTime));
        }

        while(!unAdmittedJobQueue.isEmpty()){
            while(!unAdmittedJobQueue.isEmpty() && unAdmittedJobQueue.peek().requestTime <= currentTime){
                readyJobQueue.add(unAdmittedJobQueue.poll());
            }

            if(readyJobQueue.isEmpty()){
                readyJobQueue.add(unAdmittedJobQueue.poll());
                Job job = readyJobQueue.poll();
                answer += job.requiredTime;
                currentTime = job.requestTime + job.requiredTime;
            }
            else{
                Job job = readyJobQueue.poll();
                answer += (currentTime - job.requestTime) + job.requiredTime;
                currentTime += job.requiredTime;
            }
        }

        while (!readyJobQueue.isEmpty()){
            Job job = readyJobQueue.poll();
            answer += (currentTime - job.requestTime) + job.requiredTime;
            currentTime += job.requiredTime;
        }

        return answer / jobs.length;
    }
}
```

### 모범 답안  
- 시간복잡도  
    - O(NlogN)  
- 공간복잡도  
    - O(N)
