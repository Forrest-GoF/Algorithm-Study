```java
import java.io.*;
import java.util.*;

class Solution {

    static String numbers;
    static boolean[] visited;
    static StringBuilder numberPath;
    static Set<String> primes = new HashSet<>();

    public int solution(String inputNumbers) {
        numbers = inputNumbers;
        visited = new boolean[inputNumbers.length()];
        numberPath = new StringBuilder(inputNumbers);

        for(int i = 1; i <= inputNumbers.length(); i++){
            permutation(i, 0);
        }

        return primes.size();
    }

    private void permutation(int r, int idx){
        if(idx == r){
            String number = numberPath.substring(0, r);
            if(isValidNumber(number) && isPrime(number) && !primes.contains(number)){
                primes.add(number);
            }
        }
        else{
            for(int i = 0; i < numbers.length(); i++){
                if(!visited[i]) {
                    visited[i] = true;
                    numberPath.setCharAt(idx, numbers.charAt(i));
                    permutation(r, idx + 1);
                    visited[i] = false;
                }
            }
        }
    }

    private boolean isValidNumber(String number){
        if(number.length() > 1 && number.startsWith("0")){
            return false;
        }
        return true;
    }

    private boolean isPrime(String strNumber){
        int number = Integer.valueOf(strNumber);

        if(number < 2){
            return false;
        }

        for(int i = 2; i * i <= number; i++){
            if(number % i == 0){
                return false;
            }
        }

        return true;
    }
}
```

### 모범 답안  
- 시간복잡도  
    - O(N^N * sqrt(k)) (k는 각 DFS에서의 숫자)
- 공간복잡도  
    - O(N)
